1.SELECT dept.dname,dept.deptno
FROM dept 
LEFT JOIN emp
ON(emp.deptno=dept.deptno)
WHERE emp.deptno IS NULL

2.SELECT ename,job
FROM emp
WHERE MGR IS NULL

3.SELECT emp.ename,emp.job
FROM emp, dept
WHERE emp.deptno=dept.deptno AND dept.loc LIKE 'NEW YORK' AND emp.comm>10

4.SELECT COUNT(emp.empno)
FROM emp, dept
WHERE emp.deptno=dept.deptno AND dept.loc LIKE 'CHICAGO'

5.SELECT emp.ename
FROM emp, dept
WHERE emp.deptno=dept.deptno AND dept.loc LIKE 'CHICAGO'

6.SELECT emp.ename,dept.loc
FROM emp, dept
WHERE emp.deptno=dept.deptno 
ORDER BY dept.loc

7.SELECT COUNT(empno) AS ECOUNT ,COUNT(DISTINCT deptno) AS DCOUNT
FROM emp

8.SELECT e.ename
FROM emp e,emp m
WHERE e.mgr = m.empno AND e.hiredate<m.hiredate

9.SELECT COUNT(job)
FROM emp
WHERE job LIKE 'MANAGER'

12.SELECT emp.ename,emp.job
